###Modify krb5.conf  
###set KDC and Admin Server to the KDC hosts
###set domain_realms to the dns realm
###example:  Realm = VLABKDC.LOCAL, DNS = demo.local

vi /etc/krb5.conf

[logging]
 default = FILE:/var/log/krb5libs.log
 kdc = FILE:/var/log/krb5kdc.log
 admin_server = FILE:/var/log/kadmind.log

[libdefaults]
 default_realm = VLABKDC.LOCAL
 dns_lookup_realm = false
 dns_lookup_kdc = false
 ticket_lifetime = 24h
 renew_lifetime = 7d
 forwardable = true


[realms]
 VLABKDC.LOCAL = {
  kdc = ldap-kdc.demo.local
  admin_server = ldap-kdc.demo.local
  default_domain = demo.local
 }

[domain_realm]
  .demo.local = VLABKDC.LOCAL
  demo.local = VLABKDC.LOCAL
